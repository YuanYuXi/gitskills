unsigned long  Fact(unsigned int  n){
    if(n==1||n==0){
        return 1;
    }
    else{
        return n*Fact(n-1);
    }


}

            if(a[i][i]==a[i][j]){
                sum=sum+a[i][i];

/*  函数功能： 计算n×n矩阵中两条对角线上的元素之和 */
int AddDiagonal(int a[N][N], int n)
{  	   	 	   	  
    int i, j, sum = 0; //1
    for (i = 0; i < n; i++) //1
    { 
        for (j = 0; j < n; j++) //1
        {  	   	 	   	  
            if (i == j || i + j == n - 1) //1
                sum = sum + a[i][j];
        }
    }
    return sum;
}
